# Polymarket 套利机器人配置
# 复制此文件为 .env 并填入你的配置

# ========== 钱包配置 ==========
# 你的钱包私钥（不要泄露！）
PRIVATE_KEY=your_private_key_here

# 代理钱包地址（用于交易的钱包）
PROXY_WALLET=your_wallet_address_here

# ========== API 配置 ==========
# Polymarket CLOB API
CLOB_HTTP_URL=https://clob.polymarket.com

# Polygon RPC URL - 非常重要！
# 公共 RPC 可能不稳定，建议使用专业 RPC 服务：
# - Alchemy: https://alchemy.com （免费 300M 请求/月）
# - Infura: https://infura.io （免费 100K 请求/天）
# - QuickNode: https://quicknode.com
# 示例: https://polygon-mainnet.g.alchemy.com/v2/YOUR_API_KEY
RPC_URL=https://polygon-rpc.com

# ========== 套利配置 ==========
# 最小套利空间（百分比）- 低于此值不执行
MIN_ARBITRAGE_PERCENT=0.5

# 最小单笔下单金额（美元）- 深度不够这个金额就不下单
MIN_ORDER_SIZE_USD=1

# 最大单笔下单金额（美元）- 参考trader的$1-15范围
MAX_ORDER_SIZE_USD=15

# 深度使用比例（%）- 使用订单簿可用深度的百分比
DEPTH_USAGE_PERCENT=80

# 扫描间隔（毫秒）- WebSocket 模式下主循环间隔
# 注意：订单簿数据通过 WebSocket 实时推送，这个只影响检测频率
SCAN_INTERVAL_MS=10

# ========== 单边价格阈值 ==========
# Up 价格低于此值时优先买入（参考trader的行为）
UP_PRICE_THRESHOLD=0.30

# Down 价格低于此值时优先买入
DOWN_PRICE_THRESHOLD=0.70

# ========== 并行下单 ==========
# 最多同时在几个市场下单
MAX_PARALLEL_TRADES=5

# ========== 性能配置 ==========
# 市场列表缓存时间（毫秒）- 市场列表不需要实时更新
MARKET_CACHE_MS=5000

# 并行请求订单簿数量 - 越大越快，但可能触发限流
PARALLEL_ORDERBOOK_REQUESTS=10

# ========== Telegram 通知 ==========
# Telegram 机器人 Token
TELEGRAM_BOT_TOKEN=your_bot_token_here

# Telegram 群组 ID（可以是 @群组名 或 数字ID）
TELEGRAM_GROUP_ID=@your_group_id

# 是否启用 Telegram 通知
TELEGRAM_ENABLED=true

# ========== 安全配置 ==========
# 每日最大交易次数
MAX_DAILY_TRADES=100

# 每日最大亏损（美元）- 达到后停止交易
MAX_DAILY_LOSS_USD=50

# 是否启用模拟模式（不真实下单）
SIMULATION_MODE=true

# ========== 数据存储 ==========
# 默认使用本地 JSON 文件存储（无需配置）
# 数据文件位置: ./data/bot-data.json
# 
# 如需使用 MongoDB（可选）:
# MONGODB_URI=mongodb://localhost:27017/poly-arb-bot
